```html
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <title>js中对String去空格</title>
    <style type="text/css">
    body {
        margin: 0;
        padding: 0;
    }
    </style>
</head>

<body>
    <script src="https://cdn.bootcss.com/jquery/3.3.1/jquery.min.js"></script>
    <script type="text/javascript">
    $(function() {
        var myAction = {};
        $.extend(myAction, {
            test: function() {
                var str = '  x u tong bao  ';
                var temp = str.replace(/\s+/g, ''); //去除所有空格
                console.log(':' + temp + '.');
                var temp = str.replace(/^\s+|\s+$/g, ''); //去除两头空格
                console.log(':' + temp + '.');
                var temp = str.replace(/^\s+/, ''); //去除左空格
                console.log(':' + temp + '.');
                var temp = str.replace(/(\s+$)/g, ''); //去除右空格
                console.log(':' + temp + '.');

                var temp = $.trim(str); //使用jQuery 去除两头空格
                console.log(':' + temp + '.');
            }
        });

        var init = function() {
            myAction.test();
        }();
    })
    </script>
</body>

</html>
```
