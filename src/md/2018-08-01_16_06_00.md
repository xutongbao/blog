<script type="text/javascript">
function doubleAfter2seconds(num) {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve(2 * num)
        }, 2000);
    } )
}
async function timeout() {
    var result = await doubleAfter2seconds(30);
    return result;
}
timeout().then(result => {
    console.log(result);
})
console.log('虽然在后面，但是我先执行');
</script>
