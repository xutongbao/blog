  <script>
  let args = ['xu', '徐']
  let s = '%s , %s'
  let argIndex = 0
  let newString = '%s , %s'.replace(/%s/g, () => args[argIndex++])
  console.log(s) //%s , %s
  console.log(newString) //xu , 徐
  </script>
</body>
